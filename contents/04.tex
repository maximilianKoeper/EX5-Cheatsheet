\subsubsection*{Leerstellendiffusion}

\begin{equation*}
    \begin{aligned}
        j_{diff} &= - D \nabla n_L ; \qquad n_L = \frac{N_L}{V} \\
        D &= \frac{1}{6} a^2 \nu \qquad \nu: \text{Sprungfrequenz} \\
        \nu &= \nu_0 \exp \left(- \frac{E_D}{k_B T}\right) \\
        L &= \sqrt{6Dt}
    \end{aligned}
\end{equation*}
