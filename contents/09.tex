\section{Halbleiter}

\subsubsection*{Eigenleitung}
\begin{equation*}
    \begin{aligned}
        \sigma &= e (n \mu_n + p \mu_p) \\
        \mu &= \frac{e \tau}{m^*} \\
        n &= \int_{E_L}^\infty D_L(E) f(E,T) dE \\
        p &= \int_{-\infty}^{E_V} D_V(E) (1-f(E,T)) dE
    \end{aligned}
\end{equation*}
